## JavaScript REPL

Executes JavaScript code in a sandboxed environment.

### Capabilities

| Feature | Support |
|---------|---------|
| ES2024+ syntax | ‚úÖ |
| Browser APIs | Limited |
| Console output | ‚úÖ (captured) |
| DOM access | ‚ùå (sandboxed) |
| Network requests | ‚ùå |
| Artifact access | ‚úÖ (read/write) |
| Persistent state | ‚ùå (per-call) |
| Async/await | ‚úÖ |
| npm packages | ‚ùå |

### Implementation

```typescript
interface JavaScriptREPLTool {
  name: 'javascript_repl';
  description: 'Execute JavaScript code in a sandboxed environment';
  parameters: {
    code: string;         // JavaScript code to execute
  };
}

interface JavaScriptREPLResult {
  output: string;         // Console output
  result: unknown;        // Return value (serialized)
  error?: string;         // Error message if failed
  executionTimeMs: number;
}
```

### Sandbox Environment

```typescript
interface SandboxContext {
  // Available globals
  console: {
    log: (...args: unknown[]) => void;

```

## Artifacts Tool

Creates and manages interactive content.

### Actions

| Action | Description |
|--------|-------------|
| `create` | Create new artifact |
| `update` | Update existing artifact |
| `delete` | Remove artifact |
| `list` | List all artifacts |

### Artifact Types

| Type | Extension | Renderer |
|------|-----------|----------|
| HTML | `.html` | Sandboxed iframe |
| SVG | `.svg` | Inline SVG or iframe |
| Markdown | `.md` | Rendered markdown |
| JSON | `.json` | Syntax-highlighted with folding |
| Text | `.txt` | Pre-formatted text |
| CSV | `.csv` | Table view |
| Image | `.png`, `.jpg`, `.gif` | Image element |

### Tool Definition

```typescript
interface ArtifactsTool {
  name: 'artifacts';
  description: 'Create, update, or delete artifacts';
  parameters: {
    action: 'create' | 'update' | 'delete' | 'list';
    filename?: string;
    content?: string;
    artifactType?: ArtifactType;
  };
}

interface ArtifactsResult {
  action: string;
  filename?: string;
  success: boolean;
  message: string;
  artifacts?: string[];  // For list action
}
```

### Create Flow

```typescript
// Tool call
{
  name: 'artifacts',
  arguments: {
    action: 'create',
    filename: 'chart.html',
    artifactType: 'html',
    content: '<html>...</html>'
  }
}

// Result
{
  action: 'create',
  filename: 'chart.html',
  success: true,
  message: 'Created chart.html'
}

// UI behavior:
// 1. Add to artifacts panel
// 2. Open panel if closed
// 3. Switch to new tab
// 4. Show creation toast
```

### Update Flow

```typescript
// Tool call
{
  name: 'artifacts',
  arguments: {
    action: 'update',
    filename: 'chart.html',
    content: '<html><!-- updated --></html>'
  }
}

// Result
{
  action: 'update',
  filename: 'chart.html',
  success: true,
  message: 'Updated chart.html (v2)'
}

// UI behavior:
// 1. Increment version
// 2. Refresh content viewer
// 3. Show update toast (optional)
```

### Browser Mode Implementation

Uses CORS proxy to fetch content:

```typescript
async function extractDocument(url: string): Promise<ExtractDocumentResult> {
  const proxyUrl = settings.proxy.enabled 
    ? `${settings.proxy.url}${encodeURIComponent(url)}`
    : url;
  
  const response = await fetch(proxyUrl);
  const html = await response.text();
  
  // Extract readable content (using Readability or similar)
  const doc = new DOMParser().parseFromString(html, 'text/html');
  const article = new Readability(doc).parse();
  
  return {
    url,
    title: article?.title,
    content: article?.textContent ?? '',
    wordCount: article?.textContent.split(/\s+/).length ?? 0,
    truncated: false,
  };
}
```

  toolName: string;
  render: (result: ToolResult) => ReactNode;
  collapsedView?: (result: ToolResult) => string;
}

  ['read', {
    toolName: 'read',

  ['bash', {

## Tool Execution Events

### Event Types

```typescript
interface ToolExecutionStartEvent {
  type: 'tool_start';
  toolCallId: string;
  toolName: string;
  arguments: Record<string, unknown>;
}

interface ToolExecutionUpdateEvent {
  type: 'tool_update';
  toolCallId: string;
  partialResult: Partial<ToolResult>;
}

interface ToolExecutionEndEvent {
  type: 'tool_end';
  toolCallId: string;
  result: ToolResult;
  isError: boolean;
}
```

### Streaming Updates

```typescript
function handleToolUpdate(event: ToolExecutionUpdateEvent): void {
  const toolCall = toolCalls.get(event.toolCallId);
  if (!toolCall) return;
  
  // Merge partial result
  toolCall.partialResult = {
    ...toolCall.partialResult,
    ...event.partialResult,
  };
  
  // Trigger re-render
  updateToolCallUI(event.toolCallId);
}
```
